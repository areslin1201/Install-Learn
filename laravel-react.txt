# 安裝composer
sudo curl -s https://getcomposer.org/installer | php
sudo mv composer.phar /usr/bin/composer
sudo composer --version


# 安裝laravel包
sudo composer create-project laravel/laravel --prefer-dist laravel-react-js
cd laravel-react-js


# 安裝laravel/ui
sudo composer require laravel/ui


*https://tecadmin.net/install-latest-nodejs-and-npm-on-centos
# 安裝node js
sudo yum install -y gcc-c++ make
sudo curl -sL https://rpm.nodesource.com/setup_14.x | sudo -E bash -
sudo yum -y install nodejs


# 安裝react
sudo php artisan ui react
npm install
