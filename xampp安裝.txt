# 安裝 xampp  
  https://www.apachefriends.org/download.html  


# 安裝套件
  sudo yum install wget  
  sudo wget {download_link}  
  sudo chmod +x {installer_name}  
  sudo ./{installer_name}  


# xampp bug  
  sudo vim /opt/lampp/lampp  
  *把export LD_ASSUME_KERNEL=2.2.5这一行，修改为export LD_ASSUME_KERNEL=2.8.0
  
  
# php 指定  
  ln -s /opt/lampp/bin/php /usr/bin/php


# 開機啟動XAMPP  
  ln -s /opt/lampp/xampp /etc/rc.d/init.d/xampp  
  sudo chkconfig –add xampp  
  sudo chkconfig xampp on  
  sudo /opt/lampp/xampp restart  


# apache不要顯示目錄及檔案  
  sudo vim /opt/lampp/etc/httpd.conf  
  *Options Indexes FollowSymLinks ExecCGI Includes > Indexes 拿掉  

# xampp 綁定多個網站   
  *開啟虛擬目錄  
    sudo vim /opt/lampp/etc/httpd.conf  
    -----
    # Virtual hosts  
    Include etc/extra/httpd-vhosts.conf
    -----

  sudo vim /opt/lampp/etc/extra/httpd-vhosts.conf  
    *demo  
    -----
    <VirtualHost IPADDRESS:PORT>  
      DocumentRoot "PATH/TO/WEBISTE/ROOT"  
      ServerName WEBSITE-HOST-NAME  
      <Directory "PATH/TO/WEBISTE/ROOT">  
        Options +SymLinksIfOwnerMatch  
        Require all granted  
      </Directory>  
    </VirtualHost>
    ------
